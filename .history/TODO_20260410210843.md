### 4. `server.js` を作成する

`/srv/http/shouchan-api/server.js` に以下の内容を丸ごとコピーする。

> ⚠️ `ADMIN_TOKEN` は `src/main/index.ts` の `DEV_ADMIN_TOKEN` と**必ず同じ値**にすること。  
> ⚠️ `JWT_SECRET` は推測されにくい長いランダム文字列に変更すること。

```js
const express = require('express')
const cors = require('cors')
const path = require('path')
const fs = require('fs')
const multer = require('multer')
const bcrypt = require('bcrypt')
const jwt = require('jsonwebtoken')

const app = express()
app.use(cors())
app.use(express.json())

// ===== 設定 =====
const BASE_DIR    = '/srv/http/shouchan-api'
const ADMIN_TOKEN = 'shouchan-admin-secret-2026-0728' // ← DEV_ADMIN_TOKEN と合わせる
const JWT_SECRET  = 'your-jwt-secret-change-this'     // ← 必ず変更する
const USERS_FILE  = path.join(BASE_DIR, 'accounts.json')
const MODPACKS_FILE = path.join(BASE_DIR, 'modpacks.json') // 複数ModPack一覧

// ===== ヘルパー =====
function getUsers() {
  if (!fs.existsSync(USERS_FILE)) return []
  return JSON.parse(fs.readFileSync(USERS_FILE, 'utf8'))
}
function saveUsers(users) {
  fs.writeFileSync(USERS_FILE, JSON.stringify(users, null, 2))
}
function getModpacks() {
  if (!fs.existsSync(MODPACKS_FILE)) return []
  return JSON.parse(fs.readFileSync(MODPACKS_FILE, 'utf8'))
}
function saveModpacks(list) {
  fs.writeFileSync(MODPACKS_FILE, JSON.stringify(list, null, 2))
}
function modpackDir(id) {
  return path.join(BASE_DIR, 'modpacks', id)
}

// ===== 認証ミドルウェア =====
function adminAuth(req, res, next) {
  const token = req.headers['authorization']?.replace('Bearer ', '')
  if (token !== ADMIN_TOKEN) return res.status(401).json({ error: 'Unauthorized' })
  next()
}
function userAuth(req, res, next) {
  try {
    const token = req.headers['authorization']?.replace('Bearer ', '')
    const decoded = jwt.verify(token, JWT_SECRET)
    const users = getUsers()
    req.user = users.find(u => u.id === decoded.id)
    if (!req.user) return res.status(401).json({ error: 'Invalid token' })
    next()
  } catch { res.status(401).json({ error: 'Invalid token' }) }
}

// ===== 公開エンドポイント =====

app.get('/status', (req, res) => res.json({ status: 'ok' }))

app.get('/news', (req, res) => {
  const file = path.join(BASE_DIR, 'news.json')
  if (!fs.existsSync(file)) return res.json([])
  res.json(JSON.parse(fs.readFileSync(file, 'utf8')))
})

// --- 複数ModPack ---

// ModPack一覧
app.get('/modpack/list', (req, res) => {
  res.json(getModpacks())
})

// 個別ModPack情報
app.get('/modpack/:id/info', (req, res) => {
  const mp = getModpacks().find(m => m.id === req.params.id)
  if (!mp) return res.status(404).json({ error: 'Not found' })
  res.json(mp)
})

// 個別ModPackのファイルマニフェスト
app.get('/modpack/:id/files', (req, res) => {
  const manifestFile = path.join(modpackDir(req.params.id), 'files.json')
  if (!fs.existsSync(manifestFile)) return res.status(404).json({ error: 'Manifest not found' })
  const data = JSON.parse(fs.readFileSync(manifestFile, 'utf8'))
  res.setHeader('x-modpack-version', data.version || '')
  res.json(data.files || [])
})

// 個別ModPackのファイルダウンロード
app.use('/modpack/:id/files/download', (req, res, next) => {
  req.modpackId = req.params.id
  next()
}, (req, res) => {
  const dir = modpackDir(req.modpackId)
  express.static(path.join(dir, 'files'))(req, res, () => res.status(404).end())
})

// --- 後方互換（旧 /modpack/info, /modpack/files エンドポイント） ---
app.use('/modpack/files/download', express.static(path.join(BASE_DIR, 'modpack')))

app.get('/modpack/info', (req, res) => {
  const file = path.join(BASE_DIR, 'modpack-info.json')
  if (!fs.existsSync(file)) return res.status(404).json({ error: 'Not found' })
  res.json(JSON.parse(fs.readFileSync(file, 'utf8')))
})

app.get('/modpack/files', (req, res) => {
  const file = path.join(BASE_DIR, 'modpack-files.json')
  if (!fs.existsSync(file)) return res.status(404).json({ error: 'Not found' })
  const data = JSON.parse(fs.readFileSync(file, 'utf8'))
  res.setHeader('x-modpack-version', data.version || '')
  res.json(data.files || [])
})

// ===== アカウントAPI =====

app.post('/account/register', async (req, res) => {
  const { username, email, password } = req.body
  if (!username || !email || !password)
    return res.status(400).json({ error: '全項目を入力してください' })
  const users = getUsers()
  if (users.find(u => u.email === email))
    return res.status(400).json({ error: 'このメールアドレスは既に使用されています' })
  const hash = await bcrypt.hash(password, 10)
  const user = { id: Date.now().toString(), username, email, passwordHash: hash, role: 'player', createdAt: new Date().toISOString(), settings: {} }
  users.push(user)
  saveUsers(users)
  const token = jwt.sign({ id: user.id }, JWT_SECRET, { expiresIn: '30d' })
  res.json({ success: true, id: user.id, token, role: user.role, createdAt: user.createdAt })
})

app.post('/account/login', async (req, res) => {
  const { email, password } = req.body
  const users = getUsers()
  const user = users.find(u => u.email === email)
  if (!user || !(await bcrypt.compare(password, user.passwordHash)))
    return res.status(401).json({ error: 'メールアドレスまたはパスワードが間違っています' })
  const token = jwt.sign({ id: user.id }, JWT_SECRET, { expiresIn: '30d' })
  res.json({ success: true, id: user.id, token, username: user.username, role: user.role, createdAt: user.createdAt, settings: user.settings || {} })
})

app.get('/account/profile', userAuth, (req, res) => {
  const { passwordHash, ...safe } = req.user
  res.json(safe)
})

app.put('/account/settings', userAuth, (req, res) => {
  const users = getUsers()
  const idx = users.findIndex(u => u.id === req.user.id)
  users[idx].settings = { ...users[idx].settings, ...req.body.settings }
  saveUsers(users)
  res.json({ success: true })
})

app.get('/account/verify', userAuth, (req, res) => {
  res.json({ success: true, role: req.user.role })
})

// ===== 開発者専用エンドポイント =====

// --- 複数ModPack CRUD ---

// ModPack一覧（管理用）
app.get('/admin/modpacks', adminAuth, (req, res) => {
  res.json(getModpacks())
})

// ModPack作成
app.post('/admin/modpacks', adminAuth, (req, res) => {
  const { name, version, mcVersion, forgeVersion, description } = req.body
  if (!name || !version || !mcVersion)
    return res.status(400).json({ error: 'name, version, mcVersion は必須です' })
  const { name, version, mcVersion, modLoader, loaderVersion, description } = req.body
  const id = name.toLowerCase().replace(/[^a-z0-9]/g, '-').replace(/-+/g, '-') + '-' + Date.now()
  const mp = { id, name, version, mcVersion, modLoader: modLoader || undefined, loaderVersion: loaderVersion || undefined, description: description || undefined }
  const list = getModpacks()
  list.push(mp)
  saveModpacks(list)
  fs.mkdirSync(path.join(modpackDir(id), 'files'), { recursive: true })
  res.json({ success: true, id })
})

// ModPack情報更新
app.put('/admin/modpacks/:id', adminAuth, (req, res) => {
  const list = getModpacks()
  const idx = list.findIndex(m => m.id === req.params.id)
  if (idx === -1) return res.status(404).json({ error: 'Not found' })
  list[idx] = { ...list[idx], ...req.body }
  saveModpacks(list)
  res.json({ success: true })
})

// ModPack削除
app.delete('/admin/modpacks/:id', adminAuth, (req, res) => {
  let list = getModpacks()
  list = list.filter(m => m.id !== req.params.id)
  saveModpacks(list)
  const dir = modpackDir(req.params.id)
  if (fs.existsSync(dir)) fs.rmSync(dir, { recursive: true })
  res.json({ success: true })
})

// ModPackへのファイルアップロード
const modpackStorage = multer.diskStorage({
  destination: (req, file, cb) => {
    const dest = path.dirname(path.join(modpackDir(req.params.id), 'files', req.body.path || file.originalname))
    fs.mkdirSync(dest, { recursive: true })
    cb(null, dest)
  },
  filename: (req, file, cb) => cb(null, path.basename(req.body.path || file.originalname))
})
const uploadModpack = multer({ storage: modpackStorage })
app.post('/admin/modpacks/:id/upload', adminAuth, uploadModpack.single('file'), (req, res) => res.json({ success: true }))

// ModPackのファイル一覧
app.get('/admin/modpacks/:id/files', adminAuth, (req, res) => {
  const filesDir = path.join(modpackDir(req.params.id), 'files')
  const result = []
  function walk(d, base = '') {
    if (!fs.existsSync(d)) return
    for (const name of fs.readdirSync(d)) {
      const full = path.join(d, name)
      const rel = base ? `${base}/${name}` : name
      if (fs.statSync(full).isDirectory()) walk(full, rel)
      else result.push({ path: rel, size: fs.statSync(full).size })
    }
  }
  walk(filesDir)
  res.json(result)
})

// ModPackのファイル削除
app.delete('/admin/modpacks/:id/files', adminAuth, (req, res) => {
  const filePath = path.join(modpackDir(req.params.id), 'files', req.body.path)
  if (fs.existsSync(filePath)) fs.unlinkSync(filePath)
  res.json({ success: true })
})

// ModPackマニフェスト更新（一括アップロード後）
app.put('/admin/modpacks/:id/manifest', adminAuth, (req, res) => {
  const { version, files: fileList } = req.body
  const dir = modpackDir(req.params.id)
  fs.mkdirSync(dir, { recursive: true })
  fs.writeFileSync(path.join(dir, 'files.json'), JSON.stringify({ version, files: fileList }, null, 2))
  // ModPack一覧のバージョンも更新
  const list = getModpacks()
  const idx = list.findIndex(m => m.id === req.params.id)
  if (idx !== -1) { list[idx].version = version; saveModpacks(list) }
  res.json({ success: true })
})

// --- 旧エンドポイント（後方互換） ---

app.get('/admin/files', adminAuth, (req, res) => {
  const dir = path.join(BASE_DIR, 'modpack')
  const result = []
  function walk(d, base = '') {
    if (!fs.existsSync(d)) return
    for (const name of fs.readdirSync(d)) {
      const full = path.join(d, name)
      const rel = base ? `${base}/${name}` : name
      if (fs.statSync(full).isDirectory()) walk(full, rel)
      else result.push({ path: rel, size: fs.statSync(full).size })
    }
  }
  walk(dir)
  res.json(result)
})

const legacyStorage = multer.diskStorage({
  destination: (req, file, cb) => {
    const dest = path.dirname(path.join(BASE_DIR, 'modpack', req.body.path || file.originalname))
    fs.mkdirSync(dest, { recursive: true })
    cb(null, dest)
  },
  filename: (req, file, cb) => cb(null, path.basename(req.body.path || file.originalname))
})
const uploadLegacy = multer({ storage: legacyStorage })
app.post('/admin/upload', adminAuth, uploadLegacy.single('file'), (req, res) => res.json({ success: true }))

app.delete('/admin/files', adminAuth, (req, res) => {
  const filePath = path.join(BASE_DIR, 'modpack', req.body.path)
  if (fs.existsSync(filePath)) fs.unlinkSync(filePath)
  res.json({ success: true })
})

app.put('/admin/modpack/info', adminAuth, (req, res) => {
  fs.writeFileSync(path.join(BASE_DIR, 'modpack-info.json'), JSON.stringify(req.body, null, 2))
  res.json({ success: true })
})

app.put('/admin/modpack/manifest', adminAuth, (req, res) => {
  const { version, files: fileList } = req.body
  fs.writeFileSync(path.join(BASE_DIR, 'modpack-files.json'), JSON.stringify({ version, files: fileList }, null, 2))
  res.json({ success: true })
})

app.get('/admin/news', adminAuth, (req, res) => {
  const file = path.join(BASE_DIR, 'news.json')
  if (!fs.existsSync(file)) return res.json([])
  res.json(JSON.parse(fs.readFileSync(file, 'utf8')))
})

app.put('/admin/news', adminAuth, (req, res) => {
  fs.writeFileSync(path.join(BASE_DIR, 'news.json'), JSON.stringify(req.body, null, 2))
  res.json({ success: true })
})

app.listen(3000, () => console.log('Shouchan API running on port 3000'))
```

### 8. nginxの設定

Arch Linuxでは `sites-available` がない場合があるので `/etc/nginx/conf.d/shouchan.conf` に作成する。

```bash
sudo nano /etc/nginx/conf.d/shouchan.conf
```

内容:
```nginx
server {
    listen 80;
    server_name srgk.ddns.net;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name srgk.ddns.net;

    ssl_certificate     /etc/letsencrypt/live/srgk.ddns.net/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/srgk.ddns.net/privkey.pem;

    client_max_body_size 500M;

    location /modpack/files/download/ {
        alias /srv/http/shouchan-api/modpack/;
        autoindex off;
    }

    location ~ ^/modpack/([^/]+)/files/download/(.*)$ {
        alias /srv/http/shouchan-api/modpacks/$1/files/$2;
        autoindex off;
    }

    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 300s;
    }
}
```

```bash
sudo nginx -t
sudo systemctl reload nginx
```
